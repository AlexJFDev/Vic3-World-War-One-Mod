    flavor = general_staff_laws.1.f
    category = enactment
    event_image = { texture = "gfx/event_pictures/politics_europe.dds" }

    duration = 3
    cooldown = { days = normal_modifier_time }

			add_amendment = {

				cooldown = 120
			}
		}
    }
}

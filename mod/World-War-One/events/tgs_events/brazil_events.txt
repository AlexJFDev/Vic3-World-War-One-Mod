brazil_events.1 = { #Coronelist infighting
    type = country_event

    event_image = {
        texture = "gfx/event_pictures/first_republic_brazil.dds"
    }

    title = brazil_events.1.t
    desc = brazil_events.1.d
    flavor = brazil_events.1.f

    icon = "gfx/interface/icons/event_icons/event_protest.dds"
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    duration = 3

    trigger = {}

    immediate = {
        if = {
			limit = {
				NOT = {
					any_political_movement = {
						is_political_movement_type = movement_tenentist
					}
				}
			}
			create_political_movement = { type = movement_tenentist }
		}
    }

    option = {
        name = brazil_events.1.a
        default_option = yes
        active_law:lawgroup_centralization ?= {
			add_amendment = {
				type = coronelismo
				sponsor = ig_landowners
				cooldown = 1
			}
		}
        add_modifier = {
            name = coronelist_infighting
            years = 10
            is_decaying = yes
        }
        add_modifier = {
            name = lower_strata_nerf
            years = 20
            is_decaying = yes
        }
        trigger_event = {
            id = dev_events.26
            days = 30
        }
        s:STATE_PARANA.region_state:BRZ = {
			add_modifier = {
               name = contestado_turmoil
               years = 10
               is_decaying = yes
            }
		}
        s:STATE_SANTA_CATARINA.region_state:BRZ = {
			add_modifier = {
               name = contestado_turmoil
               years = 10
               is_decaying = yes
            }
		}
        random_political_movement = {
			limit = {
				is_political_movement_type = movement_tenentist
			}
			save_scope_as = tenentist_movement_scope
		}
		scope:tenentist_movement_scope = {
			add_modifier = {
				name = political_movement_buff_2
				years = 40
			} 
		} 
    }
}

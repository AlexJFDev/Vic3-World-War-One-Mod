    type = country_event

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    
    duration = 1
    
    event_image = {
        texture = "gfx/event_pictures/_____.dds"
    }

    trigger = {}

    option = {
        name = germany_events.4.a
        create_country = { 
			origin = root
			tag = BAV
			state = s:STATE_BAVARIA.region_state:GER
			on_created = {
				activate_law = law_type:law_council_republic
                activate_law = law_type:law_universal_suffrage
                activate_law = law_type:law_right_of_assembly
                activate_law = law_type:law_command_economy
                activate_law = law_type:law_collectivized_agriculture
                activate_law = law_type:law_womens_suffrage
                activate_law = law_type:law_state_religion	
                create_diplomatic_play = {
		           target_country = c:GER
		           type = dp_annex_war
	               war = yes
	                add_war_goal = {
		               holder = c:GER
		               type = annex_country
		               target_country = c:BAV
                    }
	            }
			}
		}
        c:GER = {
            change_infamy = -100
        }
        c:BAV = {
            change_infamy = -100
        }
        s:STATE_FRANCONIA = {
	       random_scope_state = {
		      limit = {
			   exists = c:GER
			    owner = {
				  OR = {
					c:GER = THIS
				    }
			    }
		        }
		      set_state_owner = c:BAV
            }
        }
        s:STATE_SAXONY = {
	       random_scope_state = {
		      limit = {
			   exists = c:GER
			    owner = {
				  OR = {
					c:GER = THIS
				    }
			    }
		        }
		      set_state_owner = c:BAV
            }
        }
        s:STATE_BRANDENBURG = {
			set_owner_of_provinces = {
				country = c:BAV
				provinces = { x83B1A8 xF8D457 xA040A0 }
			}
		}
        s:STATE_ELBE = {
			set_owner_of_provinces = {
				country = c:BAV
				provinces = { xC08061 }
			}
		}
        s:STATE_RUHR = {
			set_owner_of_provinces = {

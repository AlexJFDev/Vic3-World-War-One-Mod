turkey_events.8 = { #Arab uprising
    type = country_event

    title = turkey_events.8.t
    desc = turkey_events.8.d
    flavor = turkey_events.8.f
    
    icon = "gfx/interface/icons/event_icons/event_protest.dds"
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    
    duration = 3
    
    event_image = {
        texture = "gfx/event_pictures/arab_revolt.dds"
    }

    trigger = {}

    immediate = {
        create_country = {
	       tag = HDJ
	       origin = this
	       state = s:STATE_HEDJAZ.region_state:TUR
           on_created = { 
           activate_law = law_type:law_monarchy
           activate_law = law_type:law_census_voting
           activate_law = law_type:law_freedom_of_conscience
           activate_law = law_type:law_interventionism
           activate_law = law_type:law_right_of_assembly
           }
		}
        s:STATE_YEMEN = {
           random_scope_state = {
		      limit = {
			   exists = c:TUR
			    owner = {
				  OR = {
					c:TUR = THIS
				    }
			    }
		        }
		      set_state_owner = c:HDJ
            }
        }
    }

    option = { 
        name = turkey_events.8.a
        default_option = yes
        c:HDJ = {
            trigger_event = {
                id = dev_events.28
                days = 5
            }
            create_diplomatic_play = {
            name = arab_revolt

            target_country = c:TUR

            type = dp_independence

            escalation = 85
            
            add_war_goal = {
                holder = c:TUR
                type = annex_country
                target_country : c:HDJ
            }
            
            add_initiator_backers = {
                c:GBR
            }
            add_target_backers = {
                c:GER
            }
            }
        }
        ai_chance = {
		  base = 10
	    }
    }

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    
    duration = 1

    event_image = {
        texture = "gfx/event_pictures/warlord_china.dds"
    }

    trigger = {}

    option = {
        name = china_events.6.a
        default_option = yes
        escalation = 30
        create_diplomatic_play = {
		      target_country = c:JSU

		        holder = c:JSU

            }
	    }
        ai_chance = {
		base = 100
	    }
        c:ZHI = {
		trigger_event = {
			id = china_events.9
			days = 415
		}
	    }
	}

    option = {
        name = china_events.6.b
    }
}

# Romanian Intervention part 2
balkan_war_events.2 = {
	type = country_event
	
	event_image = {
        texture = "gfx/event_pictures/second_balkan_war.dds"
    }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	title = balkan_war_events.2.t
	desc = balkan_war_events.2.d
	flavor = balkan_war_events.2.f

	duration = 1

	trigger = {} # Triggered in global

    immediate = {
        random_diplomatic_play = {
           limit = {
			  initiator = { c:BUL ?= this }
              target = { c:GRE ?= this }
		    }
           save_scope_as = balkan_war_diplo_play
        }
    }

	option = {
		name = balkan_war_events.2
        default_option = yes
		scope:balkan_war_diplo_play = {
			add_target_backers = {
				c:ROM
			}
			add_war_goal = {
                holder = c:ROM
                type = conquer_state
                target_state = s:STATE_DOBRUDJA.region_state:BUL
            }
			add_war_goal = {
                holder = c:BUL
                type = conquer_state
                target_state = s:STATE_DOBRUDJA.region_state:ROM
            }
		}
		ai_chance = {
			base = 1
		}
	}

	option = {
		name = balkan_war_events.2
		s:STATE_DOBRUDJA = {
            random_scope_state = {
		       limit = {
			      exists = c:BUL
			        owner = {
				        OR = {
					      c:BUL = THIS
				        }
			        }
		        }
		        set_state_owner = c:ROM
			    add_claim = c:BUL
            }
        }
		ai_chance = {
			base = 0
		}
	}
}

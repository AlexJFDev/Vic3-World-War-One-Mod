    option = {
        name = russia_events.8.a 
        create_country = { #White Russia
	       tag = SIB
	       origin = this
	       state = s:STATE_TOMSK.region_state:RUS
           on_created = { 
           activate_law = law_type:law_monarchy
           activate_law = law_type:law_oligarchy
           activate_law = law_type:law_mass_conscription
	       create_diplomatic_play = {
		      target_country = c:RUS
		      type = dp_annex_war
	          war = yes
	          add_war_goal = {
		        holder = c:RUS
		        type = annex_country
		        target_country = c:SIB
              }
	        }
	         change_infamy = -100
          }
        }
        every_state = {
	      limit = {
		    exists = c:RUS
		    owner = {
			OR = {
				c:RUS = THIS 
			}
		   }
		   OR = {
	          NOT = {
			   region = sr:region_belarus 
			   region = sr:region_russia 
		       }
		    }
          }
	        set_state_owner = c:SIB
        }
        c:RUS = {
         trigger_event = {
          id = russia_events.9
          days = 5
            }
        }   
    }

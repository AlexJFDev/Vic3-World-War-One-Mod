			   exists = c:GER

					c:GER = THIS
				    }
			    }
		        }
		        set_state_owner = c:FRA
            }
        }
        c:GER = {
		    remove_diplomatic_pact = {

		    }

			  country = c:LUX

		    }

			  country = c:NET
			  type = protectorate
		    }	
            remove_diplomatic_pact = {
			  country = c:GBR
			  type = protectorate
		    }		
            every_state = {
			if = {
				limit = {
					owner = c:GER
					has_building = building_barrack
					is_capital = no
				}
				remove_building = building_barrack
			}
			if = {
				limit = {
					owner = c:GER
					has_building = building_naval_base

				}
				remove_building = building_naval_base

				limit = {

				}

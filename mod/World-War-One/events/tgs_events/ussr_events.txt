ussr_events.9 = { #Baltic conquest
    type = country_event

    title = ussr_events.9.t
    desc = ussr_events.9.d
    flavor = ussr_events.9.f
    
    icon = "gfx/interface/icons/event_icons/event_protest.dds"
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    
    duration = 3
    
    event_image = {
        texture = "gfx/event_pictures/red_army.dds"
    }

    trigger = {}

    immediate = {
        c:GER = {
            remove_diplomatic_pact = {
			  country = c:EST
			  type = protectorate
		    }
            remove_diplomatic_pact = {
			  country = c:LAT
			  type = protectorate
		    }
            remove_diplomatic_pact = {
			  country = c:LIT
			  type = protectorate
		    }
        }
    }

    option = {
        name = ussr_events.9.a
        default_option = yes
        create_diplomatic_play = {
            name = baltic_conquest

            target_country = c:EST

            type = dp_annex_war

            escalation = 85
            
            add_war_goal = {
                holder = c:EST
                type = regime_change
                target_country = c:RUS
            }
            add_war_goal = {
                holder = c:RUS
                type = annex_country
                target_country = c:LAT
            }
            add_war_goal = {
                holder = c:RUS
                type = annex_country
                target_country = c:LIT
            }        
            
            add_target_backers = {
                c:LAT
                c:LIT
            }
        }
		ai_chance = {
			base = 1
		}
	}

	option = {
        name = ussr_events.9.b
		ai_chance = {
			base = 0
		}
	}

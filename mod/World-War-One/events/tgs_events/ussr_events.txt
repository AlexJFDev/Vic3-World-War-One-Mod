						exists = c:ITA							
						this = c:ITA								
				    }
                    or = {
						exists = c:ROM						
						this = c:ROM								
				    }
                    or = {
						exists = c:GRE					
						this = c:GRE								
				    }
			    }
			}
            add_diplomatic_play_war_support = { target = c:RUS value = -200 }   
        }
		scope:valdimir_lenin_scope = {
			set_character_as_ruler = yes
		}
		trigger_event = {

        }
		trigger_event = {
            id = ussr_events.4
            years = 5

		ai_chance = {
			base = 10
		}
    }
        
    option = {
        name = ussr_events.1.b

namespace = axis_peace_deal

axis_peace_deal.1 = {
    type = country_event

    title = axis_peace_deal.1.t
    desc = axis_peace_deal.1.d
    flavor = axis_peace_deal.1.f
    
    icon = "gfx/interface/icons/event_icons/event_protest.dds"
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    
    duration = 1
    
    immediate = {}

    event_image = { 
		texture = "gfx/event_pictures/france_surrender.dds"
	}

    trigger = {}

    immediate = {
        random_diplomatic_play = {
			limit = { 
              is_diplomatic_play_type = dp_world_war 
            }
            save_scope_as = world_war_two_diplo_play
        }
        create_country = { #Free France
	       tag = FFR
	       origin = this
	       state = s:STATE_GUAYANA.region_state:FRA
            on_created = { 
             activate_law = law_type:law_presidential_republic
             activate_law = law_type:law_oligarchy
             activate_law = law_type:law_professional_army
             activate_law = law_type:law_military_oversight
             activate_law = law_type:law_right_of_assembly
             activate_law = law_type:law_interventionism
            }
        }
        c:FRA = {
            remove_diplomatic_pact = {
			  country = c:EQA
			  type = colony
		    }
            remove_diplomatic_pact = {
			  country = c:CMR
			  type = colony
		    }
        }
        random_scope_character = {
			limit = {
				has_variable = charles_de_gaulle
			}
			kill_character = yes
		}
    }

    option = {
        name = axis_peace_deal.1.a
        set_global_variable = ww2_france_defeat
        s:STATE_ILE_DE_FRANCE = {
            random_scope_state = {
		      limit = {
			   exists = c:GER
			    owner = {
				  OR = {
					c:GER = THIS
				    }
			    }
		        }
		        set_state_owner = c:FRA
            }
        }
        s:STATE_ALSACE_LORRAINE = {
            random_scope_state = {
		      limit = {
			   exists = c:FRA
			    owner = {
				  OR = {
					c:FRA = THIS
				    }
			    }
		        }
		        set_state_owner = c:GER
            }
        }
        c:GER = {
		    create_diplomatic_pact = {
			  country = c:FRA
			  type = protectorate
		    }	
        }
        scope:world_war_two_diplo_play = {
			add_initiator_backers = {
                who = c:FRA
            }
            add_target_backers = {
                who = c:FFR
            }   
            add_war_goal = {
                holder = c:FRA
                type = annex_country
				target_country = c:FFR
            }
		}
        c:FRA = {
            set_capital = STATE_ILE_DE_FRANCE
            activate_law = law_type:law_single_party_state
            activate_law = law_type:law_outlawed_dissent
            trigger_event = {
              id = dev_events.87
              days = 5
            }
            trigger_event = {
              id = dev_events.47
              days = 5
            }
            trigger_event = {
              id = dev_events.73
              days = 5
            }
            trigger_event = {
              id = decolonization_events.1
              years = 6
            }
        }
        c:FFR = {
            trigger_event = {
              id = dev_events.1
              days = 5
            }
            trigger_event = {
              id = dev_events.117
              days = 15
            }
            create_diplomatic_pact = {
			  country = c:EQA
			  type = puppet
		    }
            create_diplomatic_pact = {
			  country = c:CMR
			  type = puppet
		    }
        }
        c:EQA = {
            trigger_event = {
              id = dev_events.83
              days = 5
            }
            trigger_event = {
              id = dev_events.148
              days = 5
            }
        }
        c:CMR = {
            trigger_event = {
              id = dev_events.83
              days = 5
            }
            trigger_event = {
              id = dev_events.148
              days = 5
            }
        }
        if = {
            limit = { c:NET = { has_variable = allies_participant } }
            c:GER = {
                create_diplomatic_pact = {
			     country = c:NET
			     type = protectorate
		        }	
            }
            c:NET = {
                remove_diplomatic_pact = {
			      country = c:DEI
			      type = chartered_company
		        }
                remove_diplomatic_pact = {
			      country = c:DEI
			      type = dominion
		        }
                remove_diplomatic_pact = {
			      country = c:DEI
			      type = colony
		        }
            }
            c:DEI = {
                trigger_event = {
                  id = dev_events.15
                  days = 5
                }
            }
        }
        if = {
            limit = { c:BEL = { has_variable = allies_participant } }
            c:GER = {
                create_diplomatic_pact = {
			     country = c:BEL
			     type = protectorate
		        }	
            }
            c:BEL = {
                remove_diplomatic_pact = {
			      country = c:KON
			      type = dominion
		        }
                remove_diplomatic_pact = {
			      country = c:KON
			      type = colony
		        }
            }
        }
        if = {
            limit = {
                NOT = {
                    c:JAP = { has_variable = allies_participant }
                }
            }
            trigger_event = {
                id = axis_peace_deal.5
                days = 5
            }
        }
    }
}

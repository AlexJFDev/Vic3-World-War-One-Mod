uk_events.6 = {#Irish question JE initiation
    type = country_event

    title = uk_events.6.t
    desc = uk_events.6.d
    flavor = uk_events.6.f
    
    icon = "gfx/interface/icons/event_icons/event_protest.dds"
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    
    duration = 3
    
    immediate = {
		if = {
			limit = {
				NOT = {
					any_political_movement = {
						is_political_movement_type = movement_cultural_minority
					}
				}
			}
			create_political_movement = { type = movement_cultural_minority }
		}
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_cultural_minority
			}
			save_scope_as = irish_movement_scope
		}
	}

    event_image = {
        texture = "gfx/event_pictures/house_of_commons.dds"
    }

    trigger = {}

    option = {
        name = uk_events.6.a
        default_option = yes
		scope:irish_movement_scope = {
			add_modifier = {
				name = political_movement_buff_2
				years = 10
			}
		}   
        trigger_event = {
            id = uk_events.7
            years = 3
        }
        ai_chance = {
			base = 8
		}
    }

    option = {
        name = uk_events.6.b
        if = {
            has_law = law_type:law_unitary
            activate_law = law_type:law_devolved
        }
        if = {
            has_law = law_type:law_devolved
            activate_law = law_type:law_federated
        }
        if = {
            has_law = law_type:law_federated
            activate_law = law_type:law_confederated
        }
        ai_chance = {
			base = 2
		}
    }

    option = {
        name = uk_events.6.c
        create_country = {
	       tag = IRE
	       origin = this
	       state = s:STATE_LEINSTER.region_state:GBR
           on_created = { 
           activate_law = law_type:law_monarchy
           }
        }
        s:STATE_MUNSTER = {
	       random_scope_state = {
		      limit = {
			   exists = c:GBR
			    owner = {
				  OR = {
					c:GBR = THIS
				    }
			    }
		        }
		      set_state_owner = c:IRE
            }
        }
        s:STATE_CONNAUGHT = {
	       random_scope_state = {
		      limit = {
			   exists = c:GBR
			    owner = {
				  OR = {
					c:GBR = THIS
				    }
			    }
		        }
		      set_state_owner = c:IRE
            }
        }
        c:GBR = {
           create_diplomatic_pact = {
			  country = c:IRE
			  type = personal_union
		    }
        }
        c:IRE = {
            trigger_event = {
             id = stabilization_buff_events.8
             days = 5
            }
        }
        ai_chance = {
			base = 0
		}
    }
}

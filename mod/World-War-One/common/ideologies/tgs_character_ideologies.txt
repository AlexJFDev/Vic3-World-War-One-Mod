ideology_rooseveltian_progressivism = {

	character_ideology = yes

    icon = "gfx/interface/icons/ideology_icons/ideology_leader/ideology_leader_nihilist.dds"

	lawgroup_economic_system = {
        law_interventionism = strongly_approve
        law_laissez_faire = disapprove
		law_command_economy = strongly_disapprove
		law_traditionalism = strongly_disapprove
		law_agrarianism = strongly_disapprove
		law_cooperative_ownership = strongly_disapprove
		law_industry_banned = strongly_disapprove
		law_extraction_economy = strongly_disapprove
	}
 
    lawgroup_taxation = {
        law_graduated_taxation = approve
        law_proportional_taxation = disapprove
		law_consumption_based_taxation = disapprove
		law_per_capita_based_taxation = strongly_disapprove
		law_land_based_taxation = strongly_disapprove
	}

    lawgroup_education_system = {
        law_public_schools = strongly_approve
		law_private_schools = disapprove
		law_religious_schools = strongly_disapprove
		law_no_schools = strongly_disapprove
	}
    
	lawgroup_health_system = {
        law_public_health_insurance = approve
		law_private_health_insurance = neutral
		law_charitable_health_system = neutral
		law_no_health_system = strongly_disapprove
	}
	
	lawgroup_labor_rights = {
        law_no_workers_rights = strongly_disapprove
		law_regulatory_bodies = neutral
		law_worker_protections = approve
		law_limited_work_hours = strongly_approve
	}

    lawgroup_childrens_rights = {
        law_child_labor_allowed = strongly_disapprove
		law_restricted_child_labor = neutral
		law_compulsory_primary_school = strongly_approve
	}

    lawgroup_welfare = {
        law_no_social_security = strongly_disapprove
		law_poor_laws = neutral
		law_wage_subsidies = approve
		law_old_age_pension = strongly_approve
	}

	lawgroup_labour_associations = {
		law_combination_acts = strongly_disapprove
		law_guild_system = strongly_disapprove
		law_anti_strike_laws = disapprove
		law_corporatized_unions = strongly_approve
		law_right_to_associate = approve
		law_factory_councils = strongly_disapprove
	}

	country_trigger = {
		country_has_primary_culture = cu:euro_american
	}

	interest_group_leader_trigger = {
		ideology_rooseveltian_progressivism_valid_trigger = yes
	}
    
	non_interest_group_leader_trigger = {
		interest_group ?= {
			ideology_rooseveltian_progressivism_valid_trigger = yes
		}
	}

	interest_group_leader_weight = {
		value = {
			add = 100
			desc = "base_value"
		}

		# more likely for intelligentsia
		if = {
			limit = {
				is_interest_group_type = ig_intelligentsia
			}
			value = {
				add = 50
				desc = "ig_intelligentsia"
			}
		}

		# More likely if issues are relevant - greater bonus to this than usual
		if = {
			limit = {
				owner.ideology_humanitarian_disfavoured_laws > 0
			}
			add = {
				value = owner.ideology_humanitarian_disfavoured_laws
				multiply = 16
				desc = "DISFAVORED_LAWS"
			}
		}

		# Less likely if issues are not relevant
		if = {
			limit = {
				owner.ideology_humanitarian_favoured_laws > 0
			}
			subtract = {
				value = owner.ideology_humanitarian_favoured_laws
				multiply = 8
				desc = "FAVORED_LAWS"
			}
		}

		# less likely for angry trade unions outside of Council Republic, they should go communist instead
		if = {
			limit = {
				owner = {
					has_technology_researched = socialism
					NOT = {
						has_law_or_variant = law_type:law_council_republic
					}
				}
				is_interest_group_type = ig_trade_unions
				ig_approval <= define:NPolitics|APPROVAL_THRESHOLD_UNHAPPY
			}
			value = {
				add = -50
				desc = "IG_APPROVAL_BELOW_UNHAPPY"
			}
		}

		if = {
			limit = {
				OR = {
					has_variable = communist_ig_var
					has_variable = chose_vanguardism
					has_variable = chose_anarchism
					has_variable = chose_communism
					has_variable = chose_ethno_nationalism
					has_variable = chose_conservatism
					has_variable = chose_fascism
				}
			}
			multiply = {
				value = 0.1
				desc = "CHOSE_A_DIFFERENT_PATH"
			}
		}
	}

    non_interest_group_leader_weight = {
		value = 100

		# more likely for intelligentsia
		if = {
			limit = {
				is_interest_group_type = ig_intelligentsia
			}
			add = 50
		}

		# More likely if issues are relevant - greater bonus to this than usual
		if = {
			limit = {
				owner.ideology_humanitarian_disfavoured_laws > 0
			}
			add = {
				value = owner.ideology_humanitarian_disfavoured_laws
				multiply = 16
			}
		}

		# Less likely if issues are not relevant
		if = {
			limit = {
				owner.ideology_humanitarian_favoured_laws > 0
			}
			subtract = {
				value = owner.ideology_humanitarian_favoured_laws
				multiply = 8
			}
		}

		# less likely for angry trade unions outside of Council Republic, they should go communist instead
		if = {
			limit = {
				owner = {
					has_technology_researched = socialism
					NOT = {
						has_law_or_variant = law_type:law_council_republic
					}
				}
				is_interest_group_type = ig_trade_unions
				interest_group.ig_approval < 5
			}
			add = -50
		}

		if = {
			limit = {
				interest_group ?= {
					OR = {
						has_variable = chose_ethno_nationalism
						has_variable = chose_conservatism
						has_variable = chose_fascism
					}
				}
			}
			multiply = 0.1
		}
		else_if = {
			limit = {
				interest_group ?= {
					OR = {
						has_variable = communist_ig_var
						has_variable = chose_vanguardism
						has_variable = chose_anarchism
						has_variable = chose_communism
					}
				}
			}
			multiply = 0.5
		}
	}
}

			else_if = {
				limit = {
					is_interest_group_type = ig_trade_unions
					owner = {
						OR = {
							has_law_or_variant = law_type:law_guild_system
							has_law_or_variant = law_type:law_anti_strike_laws
						}
					}
				}
				add = 25
			}
			else_if = {
				limit = {
					is_interest_group_type = ig_trade_unions
					owner = {
						has_law_or_variant = law_type:law_right_to_associate
					}
				}
				add = -25
			}
			else_if = {
				limit = {
					is_interest_group_type = ig_trade_unions
					owner = {
						has_law_or_variant = law_type:law_corporatized_unions
					}
				}
				add = -50
			}
		}
		if = {
			limit = {
				owner = {
					has_global_variable = peoples_springtime_happening
					capital = {
					 	OR = {
							is_in_geographic_region = geographic_region_north_germany_old
							is_in_geographic_region = geographic_region_south_germany_old
							is_in_geographic_region = geographic_region_england_old
							is_in_geographic_region = geographic_region_iberia_old
							is_in_geographic_region = geographic_region_france_old
							is_in_geographic_region = geographic_region_occitania_old
							is_in_geographic_region = geographic_region_rhine_old
							is_in_geographic_region = geographic_region_italy_old
							is_in_geographic_region = geographic_region_poland_old
							is_in_geographic_region = geographic_region_danubia_old
							is_in_geographic_region = geographic_region_balkans_old
							is_in_geographic_region = geographic_region_baltic_old
						}
					}
				}
			}
			multiply = {
				value = 2
			}
		}
		if = {
			limit = {
				owner = { has_modifier = post_revolutionary_fervor }
			}
			multiply = { value = 1.5 }
		}
		if = {
			limit = {
				owner = { has_modifier = counter_revolutionary_fervor }
			}
			multiply = { value = 0.5 }
		}
		# Becomes less prevalent after socialism is researched
		if = {
			limit = {
				owner = {
					has_technology_researched = socialism
				}
			}
			if = {
				limit = {
					interest_group ?= {
						is_interest_group_type = ig_trade_unions
					}
				}
				multiply = 0.05
			}
			else = {
				multiply = 0.5
			}
		}
		if = {
			limit = {
				interest_group ?= {
					OR = {
						has_variable = communist_ig_var
						has_variable = chose_vanguardism
						has_variable = chose_anarchism
						has_variable = chose_communism
						has_variable = chose_ethno_nationalism
						has_variable = chose_conservatism
						has_variable = chose_fascism
					}
				}
			}
			multiply = 0.1
		}
	}
}

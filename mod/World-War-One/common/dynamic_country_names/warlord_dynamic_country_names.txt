REPLACE_OR_CREATE:FRS = { # Min
	dynamic_country_name = {
		name = FRS_warlord
		adjective = FRS_ADJ
		
		is_main_tag_only = yes
		priority = 0
		
		trigger = {
			scope:actor ?= {
				has_law = law_type:law_presidential_republic
				has_law = law_type:law_autocracy 
				OR = {
					has_law = law_type:law_peasant_levies
					has_law = law_type:law_mass_conscription
					has_law = law_type:law_professional_army
				}
			}
		}
	}
	dynamic_country_name = {
		name = FRS_warlord_great
		adjective = FRS_ADJ
		
		is_main_tag_only = yes
		priority = 10
		
		trigger = {
			scope:actor ?= {
				has_law = law_type:law_presidential_republic
				has_law = law_type:law_autocracy 
				has_law = law_type:law_peasant_levies
				any_scope_state = {
					OR = { #Maybe theres a better trigger?
						region = sr:region_manchuria #满洲
						region = sr:region_north_china #华北
						region = sr:region_south_china #华南
						region = sr:region_central_asia #中亚
					}
					count >= 5
				}
			}
		}
	}
	dynamic_country_name = {
		name = FRS_republic
		adjective = FRS_ADJ
		
		is_main_tag_only = yes
		priority = 0
		
		trigger = {
			OR = {
				AND = {
					NOT = { has_law = law_type:law_monarchy }
					country_has_voting_franchise = yes
				}
				coa_def_communist_flag_trigger = yes
			}
		}
	}
	dynamic_country_name = {
		name = FRS_imperial
		adjective = FRS_ADJ
		
		is_main_tag_only = yes
		priority = 0
		
		trigger = {
			scope:actor ?= {
				has_law = law_type:law_monarchy
			}
		}
	}
	dynamic_country_name = {
		name = FRS_qing_governor
		adjective = FRS_ADJ
   
		is_main_tag_only = yes
		priority = 10
   
		trigger = {
			scope:actor ?= {

POLITICAL_MOVEMENTS = {
	every_country = {
		limit = { NOT = { is_country_type = decentralized } }
		### Royalist movements
		if = {
			limit = {
				has_law = law_type:law_monarchy
				country_has_voting_franchise = no
				NOT = { c:FRA ?= this } # No generic royalist movement for France
				NOT = { c:POR ?= this } # No absolutists for Portugal
				NOR = {
					c:SPA ?= this # No absolutists for the Spains
					c:SPC ?= this
				}
			}
			create_political_movement = { type = movement_royalist_absolutist }
		}
		else_if = {
			limit = {
				has_law = law_type:law_monarchy
				OR = {
					country_has_voting_franchise = yes
					c:POR ?= this
				}
				NOT = { c:FRA ?= this } # No generic royalist movement for France
			}
			create_political_movement = { type = movement_royalist_constitutional }
		}

		### Liberal movements
		if = {
			limit = {
				has_technology_researched = democracy
				OR = {
					has_law_or_variant = law_type:law_landed_voting
					has_law_or_variant = law_type:law_wealth_voting
					has_law_or_variant = law_type:law_census_voting
					has_law_or_variant = law_type:law_universal_suffrage
					has_law_or_variant = law_type:law_presidential_republic
					has_law_or_variant = law_type:law_parliamentary_republic
				}
				country_should_exclusively_focus_on_modernization = no
			}
			create_political_movement = { type = movement_liberal }
		}

		### Labour movements
		if = {
			limit = {
				has_technology_researched = labor_movement
			}
			create_political_movement = { type = movement_labor }
		}

		### Abolitionist movements
		if = {
			limit = {
				NOT = { has_law_or_variant = law_type:law_slavery_banned }
			}
			if = {
				limit = {
					NOT = {
						has_law_or_variant = law_type:law_colonial_slavery
					}
				}
				create_political_movement = { type = movement_pro_slavery }
			}
			if = {
				limit = {
					has_technology_researched = empiricism
					NOT = { c:BIC ?= this } # British India has Utilitarians
				}
				create_political_movement = { type = movement_anti_slavery }
			}
		}

		### Religious movements
		if = {
			limit = {
				c:BIC ?= this
			}
			create_political_movement = { type = movement_religious_minority religion = rel:hindu }
			create_political_movement = { type = movement_religious_minority religion = rel:sunni }
			create_political_movement = { type = movement_cultural_minority culture = cu:bengali }
		}
		else_if = { # France gets Legitimists instead, Spain gets Carlists, Portugal gets Miguelists
			limit = {
				NOR = {
					c:FRA ?= this
					c:SPA ?= this
					c:SPC ?= this
					c:POR ?= this
				}
			}
			create_political_movement = { type = movement_religious_majority }
		}

		### Cultural majority movements
		if = {
			limit = {
				country_has_primary_culture = cu:german
			}
			create_political_movement = { type = movement_german_national }
		}
		else_if = {
			limit = {
				country_has_primary_culture = cu:italian
			}
			create_political_movement = { type = movement_italian_national }
		}
		else = {
			create_political_movement = { type = movement_cultural_majority }
		}
	}
}

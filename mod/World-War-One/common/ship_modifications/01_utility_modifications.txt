### Utility Modifications
### These can be assigned to any utility slot on any ship type

# Patrol Boats

utility_mod_patrol_boats = { #Steel and engines
	type = ship_mod_slot_utility_1
	icon = "gfx/interface/icons/military_icons/navy_icons/marine_capacity.dds"

	modifier = {
        ship_blockade_strength_mult = 0.5
	}

	construction_goods = {
		goods_input_steel_add = 4
		goods_input_engines_add = 1
	}

	ai_weight = {
		value = 1
		multiply = ship_mod_market_multiplier
		multiply = ship_mod_coastal_defence_doctrine
		multiply = ship_mod_merchant_navy_doctrine
	}
}

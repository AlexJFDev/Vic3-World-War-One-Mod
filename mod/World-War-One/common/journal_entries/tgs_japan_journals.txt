je_japan_expansion_1 = {
    icon = "gfx/interface/icons/event_icons/rising_sun.dds"
    group = je_group_internal_affairs

    is_shown_when_inactive = {
        exists = c:MCH
        exists = c:JAP
        THIS = c:JAP
    }
    
    possible = {
        ruler = {
            OR = {
				has_ideology = ideology:ideology_fascist
				has_ideology = ideology:ideology_reactionary
                has_ideology = ideology:ideology_jingoist_leader
            }
        }
        exists = c:MCH
        game_date > 1925.1.1
    }

    immediate = {}

    complete = {
        c:MCH = {
            NOT = { 
                is_subject_of = c:JAP
            }
            OR = {
                is_subject = no
                is_subject_of = c:CHI
            }
        }
        NOT = {
            c:JAP = {
                any_scope_treaty = {
					binds = c:MCH
					any_scope_article = {
					OR = {
						has_type = alliance
						has_type = defensive_pact
						}
					}
				}
            }
            has_modifier = declared_bankruptcy
        }
        army_size >= 200
        game_date > 1931.8.1
        is_at_war = no
        ruler = {
            OR = {
				has_ideology = ideology:ideology_fascist
				has_ideology = ideology:ideology_reactionary
                has_ideology = ideology:ideology_jingoist_leader
            }
        }
    }

    on_complete = {
        trigger_event = { 
           id = japan_events.18
           days = 5
        }
    }

    invalid = {
        NOT = { exists = c:MCH }
    }

    should_be_pinned_by_default_uninvolved_or_context = yes
}

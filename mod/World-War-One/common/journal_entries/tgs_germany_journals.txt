je_war_fatigue = {
    icon = "gfx/interface/icons/event_icons/event_protest.dds"
    group = je_group_internal_affairs

    is_shown_when_inactive = {
        exists = c:GER
        THIS = c:GER
    }
    
    possible = {
        has_variable = central_powers_participant
        any_scope_war = {
		    has_war_support = {
		        target = ROOT
		        value < 5
		    }
		}
        root.average_sol =< 12
    }

    modifiers_while_active = {
        war_fatigue_1
    }

    immediate = {
       c:GER = {
            trigger_event = {
                id = germany_events.1
                days = 5
            }
        }
    }

    complete = {
      OR = {
           is_at_war = no
           root.average_sol >= 15
        }
    }
    
    fail = { 
        any_political_movement = {
            OR = {
				is_political_movement_type = movement_socialist
				is_political_movement_type = movement_communist
                is_political_movement_type = movement_anarchist
                is_political_movement_type = movement_radical
                is_political_movement_type = movement_labor
                is_political_movement_type = movement_nihilist
			}
            is_insurrectionary = yes
			civil_war ?= {
                OR = {
                  is_civil_war_type = revolution
                  is_civil_war_type = secession
                }
				civil_war_progress >= 0.75
			}
        }
    }

    on_fail = { 
      random_civil_war = {
			add_civil_war_progress = -100
		}
      c:GER = {
            trigger_event = {
                id = germany_events.2
                days = 5
            }
        }
	  c:AUS = {
            trigger_event = {
                id = germany_events.6
                days = 15
            }
        }
      set_variable = german_revolution_1
    }

    invalid = {
       OR = {
        c:GER = {
          has_variable = central_powers_participant
		  has_modifier = country_humiliated
        } 
        NOT = {
            owns_entire_state_region = STATE_ALSACE_LORRAINE
        }
       }
    }

    on_invalid = {
        set_global_variable = german_revolution_1
        random_civil_war = {
			add_civil_war_progress = -100
		}
        trigger_event = {
			id = germany_events.2
            days = 5
		}
	}

    timeout = -1
    should_be_pinned_by_default = yes
}

je_socialist_split = {
    icon = "gfx/interface/icons/event_icons/red_star.dds"
    group = je_group_internal_affairs

    is_shown_when_inactive = {
        exists = c:GER
        THIS = c:GER
    }
    
    possible = {
      has_global_variable = german_revolution_1
      NOT = {
        OR = {
         has_law = law_type:law_council_republic
		 has_law = law_type:law_command_economy
         has_law = law_type:law_collectivized_agriculture
        }
      }
    }

    complete = {
        ig:ig_trade_unions  = {
			ig_approval > 8
		}	
    }
    
    fail = { 
      OR = {
          any_political_movement = {
            OR = {
				is_political_movement_type = movement_socialist
				is_political_movement_type = movement_communist
                is_political_movement_type = movement_anarchist
                is_political_movement_type = movement_radical
                is_political_movement_type = movement_labor
                is_political_movement_type = movement_nihilist
			}
            is_insurrectionary = yes
			civil_war ?= {
                OR = {
                  is_civil_war_type = revolution
                  is_civil_war_type = secession
                }
				civil_war_progress >= 0.75
			}
            }
            ig:ig_trade_unions  = {
				ig_approval < -4
			}	
        }
    }

    on_complete = {
        trigger_event = {
			id = germany_events.5
            days = 5
		}
    }

    on_fail = { 
      random_civil_war = {
			add_civil_war_progress = -100
		}
      add_modifier = {
        name = revolution_stabilization_buff_3
        months = 24
        is_decaying = yes
       } 
      c:GER = {
            trigger_event = {
                id = germany_events.4
                days = 5
            }
        }
    }

    timeout = -1
    should_be_pinned_by_default = yes
}

je_fascist_threat = {
    icon = "gfx/interface/icons/event_icons/event_protest.dds"
    group = je_group_internal_affairs

    is_shown_when_inactive = {
        exists = c:GER
        THIS = c:GER
        OR = {
         has_global_variable = entente_deal_round_1_germany_dislocation_prop_1_won
         has_global_variable = entente_deal_round_1_germany_dislocation_prop_2_won
         has_global_variable = entente_deal_round_1_germany_dislocation_prop_3_won
        }
        OR = {
         has_global_variable = entente_deal_round_2_germany_war_rep_prop_2_won
         has_global_variable = entente_deal_round_2_germany_war_rep_prop_3_won
        }
    }
    
    possible = {
      OR = {
         has_global_variable = entente_deal_round_1_germany_dislocation_prop_1_won
         has_global_variable = entente_deal_round_1_germany_dislocation_prop_2_won
         has_global_variable = entente_deal_round_1_germany_dislocation_prop_3_won
        }
        OR = {
         has_global_variable = entente_deal_round_2_germany_war_rep_prop_2_won
         has_global_variable = entente_deal_round_2_germany_war_rep_prop_3_won
        }
    }

    immediate = {
       trigger_event = { id = germany_events.3 popup = yes }
       create_political_movement = { type = movement_fascist }
    }
    
    modifiers_while_active = {
	    fascist_threat_modifier
	}

    complete = {}
    
    fail = { 
      OR = {
          any_political_movement = {
            OR = {
				is_political_movement_type = movement_fascist
				is_political_movement_type = movement_reactionnary
                is_political_movement_type = movement_cultural_majority
                is_political_movement_type = movement_integralist
			}
            is_insurrectionary = yes
			civil_war ?= {
                OR = {
                  is_civil_war_type = revolution
                  is_civil_war_type = secession
                }
				civil_war_progress >= 0.75
			}
            }
            any_interest_group = {
				is_in_government = yes
				is_strongest_ig_in_government = yes
				OR = {
					has_ideology = ideology:ideology_fascist
                    has_ideology = ideology:ideology_reactionnary
                    has_ideology = ideology:ideology_authoritarian
                    has_ideology = ideology:ideology_ethno_nationalist
                    has_ideology = ideology:ideology_integralist
				  leader = {
						has_fascist_ideology = yes
						has_reactionnary_ideology = yes
                        has_authoritarien_ideology = yes
						has_ethno_nationalist_ideology = yes
                        has_integralist_ideology = yes
					}
				}
			}
        }
    }

    on_complete = {
        trigger_event = { id = germany_events.20 popup = yes }
    }

    on_fail = {
        trigger_event = { id = germany_events.21 popup = yes }
    }
      
    timeout = -1
    should_be_pinned_by_default = yes
}

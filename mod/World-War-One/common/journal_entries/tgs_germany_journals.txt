je_war_fatigue = {
    icon = "gfx/interface/icons/event_icons/event_protest.dds"
    group = je_group_internal_affairs

    is_shown_when_inactive = {
        exists = c:GER
        THIS = c:GER
        has_variable = central_powers_participant
    }
    
    possible = {
        has_variable = central_powers_participant
        average_sol < average_expected_sol
        any_scope_war = {
		    has_war_support = {
		        target = ROOT
		        value < 10
		    }
		}
    }

    modifiers_while_active = {
        war_fatigue_1
    }

    immediate = {
       c:GER = {
            trigger_event = {
                id = germany_events.1
                days = 5
            }
        }
    }

    complete = { 
        OR = {
            average_sol > average_expected_sol
            is_at_war = no
        }
    }
    
    fail = { 
        any_political_movement = {
            OR = {
				is_political_movement_type = movement_socialist
				is_political_movement_type = movement_communist
                is_political_movement_type = movement_anarchist
                is_political_movement_type = movement_radical
                is_political_movement_type = movement_labor
                is_political_movement_type = movement_nihilist
                is_political_movement_type = movement_liberal
			}
            is_insurrectionary = yes
			civil_war ?= {
                OR = {
                  is_civil_war_type = revolution
                  is_civil_war_type = secession
                }
				civil_war_progress >= 0.75
			}
        }
    }

    on_fail = { 
        random_civil_war = {
			add_civil_war_progress = -100
		}
      c:GER = {
            trigger_event = {
                id = germany_events.2
                days = 5
            }
        }
	  c:AUS = {
            trigger_event = {
                id = germany_events.6
                days = 15
            }
        }
      c:TUR = {
            trigger_event = {
                id = germany_events.9
                days = 15
            }
        }
      set_variable = german_revolution_1
    }

    invalid = {
       OR = {
        c:GER = {
          has_variable = central_powers_participant
		  has_modifier = country_humiliated
        } 
        NOT = {
            owns_entire_state_region = STATE_ALSACE_LORRAINE
        }
       }
    }

    on_invalid = {
        random_civil_war = {
			add_civil_war_progress = -100
		}
        c:GER = {
            trigger_event = {
                id = germany_events.2
                days = 5
            }
        }
	  c:AUS = {
            trigger_event = {
                id = germany_events.6
                days = 15
            }
        }
      c:TUR = {
            trigger_event = {
                id = germany_events.9
                days = 15
            }
        }
        set_global_variable = german_revolution_1
	}

    weight = 1000
    timeout = -1
    should_be_pinned_by_default = yes
}

            }
        }
    }

        ig:ig_trade_unions  = {
			ig_approval > 8
		}	
    }

      has_variable = socialist_negotiations_over

          any_political_movement = {
            OR = {
				is_political_movement_type = movement_socialist

			civil_war ?= {
                OR = {
                  is_civil_war_type = revolution
                  is_civil_war_type = secession
                }
				civil_war_progress >= 0.75
			}
            }
            ig:ig_trade_unions  = {
				ig_approval < -10
			}	
        }
    }

    on_complete = {
        trigger_event = {
			id = germany_events.5
            days = 5
		}
    }

    on_fail = { 
      random_civil_war = {
			add_civil_war_progress = -100
		}
      add_modifier = {
        name = revolution_stabilization_buff_3
        months = 24
        is_decaying = yes
       } 
      c:GER = {
            trigger_event = {
                id = germany_events.4
                days = 5
            }
        }
    }

    timeout = -1
    should_be_pinned_by_default = yes
}

				   has_ideology = ideology:ideology_fascist
                   has_ideology = ideology:ideology_authoritarian

    }

    immediate = {
        trigger_event = {
            id = germany_events.29

        average_sol > average_expected_sol
        any_political_movement = {
            is_insurrectionary = no
        }
        NOT = {
            c:GBR = {
			  has_truce_with = root
            }
            c:FRA = {
			  has_truce_with = root

			  has_truce_with = root

		    limit = {
              country_rank >= rank_value:great_power

			}

            exists = c:POL
            exists = c:CZE
        }
        is_at_war = no
    }

    on_complete = {
        trigger_event = {
            id = germany_events.35
            days = 90
        }
    }

    on_timeout = {}

    timeout = -1
    should_be_pinned_by_default = yes
}

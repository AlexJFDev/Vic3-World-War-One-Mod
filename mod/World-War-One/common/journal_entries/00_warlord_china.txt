je_warlord_china = { #TGS version
	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	group = je_group_historical_content

	is_shown_when_inactive = {
        exists = c:CHI
        THIS = c:CHI
        NOT = { has_global_variable = china_collapse }
    }
    
    possible = {
        exists = c:CHI
        THIS = c:CHI
        NOT = { has_global_variable = china_collapse }
    }

    immediate = {
       
    }

    modifiers_while_active = {
		fragile_unity_modifier
	}

    complete = {
        OR = {
			c:KOR = {
			   exists = c:KOR
			   is_subject = no
			}
			c:ICH = {
			   exists = c:ICH
			   is_subject = no
			}
			has_law = law_type:law_census_voting
            has_law = law_type:law_universal_suffrage
            has_law = law_type:law_anarchy
            has_law = law_type:law_council_republic
            has_law = law_type:law_homesteading
			has_law = law_type:law_proportional_taxation
			has_law = law_type:law_graduated_taxation 
		}
    }

    on_complete = {
		trigger_event = { id = warlord_china_events.200 }
	}

    fail = {
	  OR = {
		has_variable = yuan_shikai_abdication
		civil_war ?= {
            is_civil_war_type = revolution
			civil_war_progress >= 0.75
		}
	  }
    }

	on_fail = {
		trigger_event = { id = warlord_china_events.100 }
	}

	weight = 10000
	should_be_pinned_by_default = yes
}

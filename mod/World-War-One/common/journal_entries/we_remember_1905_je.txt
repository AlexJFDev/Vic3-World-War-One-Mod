je_russian_social_unrest = {
    icon = "gfx/interface/icons/event_icons/event_protest.dds"

    group = je_group_internal_affairs

    is_shown_when_inactive = {
        exists = c:RUS
        THIS = c:RUS
        NOT = { has_variable = russian_social_unrest_resolved }
    }

    possible = {
        exists = c:RUS
        THIS = c:RUS
        NOT = { has_variable = russian_social_unrest_resolved }
    }

    immediate = {
        c:RUS = {
            add_country_modifier = {
                name = we_remember_1905_modifier
                duration = -1 # Permanent until conditions are met
            }
        }
    }

    modifiers_while_active = {
        we_remember_1905_modifier
    }

    complete = {
        OR = {
            has_law = law_type:law_census_suffrage
            has_law = law_type:law_universal_suffrage
            has_law = law_type:law_anarchy
            has_law = law_type:law_council_republic
            has_law = law_type:law_homesteading
            has_law = law_type:law_wage_subsidies
        }
    }

    on_complete = {
        c:RUS = {
            remove_country_modifier = we_remember_1905_modifier
            set_variable = russian_social_unrest_resolved
        }
    }
    
    timeout = -1 # No timeout, this journal entry stays until completion
    should_be_pinned_by_default = yes
}

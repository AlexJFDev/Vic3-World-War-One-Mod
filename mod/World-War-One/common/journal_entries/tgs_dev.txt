je_ai_allies_landings = {
    icon = "gfx/interface/icons/invention_icons/landing_craft.dds"
    group = je_group_internal_affairs

    is_shown_when_inactive = {
       exists = c:USA
       THIS = c:USA
       is_player = no 
       has_variable = allies_participant
    }
    
    possible = {
        c:USA = {
          is_player = no
          is_at_war = yes
          has_variable = allies_participant
        }
        c:GBR = {
          is_player = no
          is_at_war = yes
          has_variable = allies_participant
        }
        c:GER = {
          is_at_war = yes
          has_variable = axis_participant
        }
        c:FRA = {
            OR = {
                AND = {
                    has_variable = axis_participant
                    is_at_war = yes
                }
                is_subject_of = c:GER
            }
        }
    }

    immediate = { }

    complete = {
        exists = c:USA
        exists = c:GBR
        exists = c:GER
        game_date > 1944.4.1
        c:USA = {
          is_player = no
          is_at_war = yes
          has_variable = allies_participant
        }
        c:GBR = {
          is_player = no
          is_at_war = yes
          has_variable = allies_participant
        }
        c:GER = {
          is_at_war = yes
          has_variable = axis_participant
        }
    }

    on_complete = {
        trigger_event = {
            id = dev_events.4
            days = 5
        }
    }

    on_timeout = {}
}

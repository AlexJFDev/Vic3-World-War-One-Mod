je_monetary_supremacy = {
    icon = "gfx/interface/icons/event_icons/event_trade.dds"
    group = je_group_internal_affairs

    is_shown_when_inactive = {
        country_rank = rank_value:great_power
    }
    
    possible = {
        global_country_ranking = 1
        navy_size >= 700
        is_at_war = no 
    }

    immediate = {
       trigger_event = { id = monetary_supremacy_events.1 popup = yes }
    }
    
    modifiers_while_active = {
        monetary_supremacy_modifier
    }

    fail = {
        NOT = {
          global_country_ranking = 1
        } 
        is_at_war = no
    }

    on_fail = { 
       trigger_event = { id = monetary_supremacy_events.2 popup = yes }
    }

    weight = 1000
    should_be_pinned_by_default = yes
}

je_irish_question = {
    icon = "gfx/interface/icons/event_icons/event_protest.dds"
    group = je_group_internal_affairs

    is_shown_when_inactive = {
        exists = c:GBR
        THIS = c:GBR
    }
    
    possible = {
		owns_entire_state_region = STATE_LEINSTER
        owns_entire_state_region = STATE_ULSTER
        owns_entire_state_region = STATE_MUNSTER
        owns_entire_state_region = STATE_CONNAUGHT
    }

    immediate = {
        trigger_event = { 
           id = uk_events.6
           days = 5
        }
    }

    fail = {
       any_political_movement = {
			is_political_movement_type = movement_cultural_minority
            is_insurrectionary = yes
			civil_war ?= {
                OR = {
                  is_civil_war_type = revolution
                  is_civil_war_type = secession
                }
				civil_war_progress >= 0.75
			}
        }
    }

    invalid = {
       NOT = {
          owns_entire_state_region = STATE_LEINSTER
          owns_entire_state_region = STATE_ULSTER
          owns_entire_state_region = STATE_MUNSTER
          owns_entire_state_region = STATE_CONNAUGHT
        }
    }

    on_fail = {
        trigger_event = { 
           id = uk_events.8
           days = 5
        }
    }

    should_be_pinned_by_default = yes
}

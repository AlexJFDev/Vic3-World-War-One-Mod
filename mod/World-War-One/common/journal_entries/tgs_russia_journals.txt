je_colossus_with_feet_of_clay = {
    icon = "gfx/interface/icons/event_icons/event_portrait.dds"
    group = je_group_internal_affairs

    is_shown_when_inactive = {
        exists = c:RUS
        THIS = c:RUS
        NOT = { has_variable = russian_reforms_completed }
    }
    
    possible = {
        has_technology_researched = socialism
        has_technology_researched = anarchism
    }

    immediate = {
       trigger_event = { id = russia_events.1 popup = yes }
    }

    modifiers_while_active = {
	   colossus_with_feet_of_clay_modifier
	}

    complete = {
        OR = {
            has_law = law_type:law_census_voting
            has_law = law_type:law_universal_suffrage
            has_law = law_type:law_anarchy
            has_law = law_type:law_council_republic
            root.average_sol >= 12
        }
    }
    
    fail = { 
        any_civil_war = {
			civil_war_progress >= 0.75
		}
    }

    on_complete = {
        trigger_event = {
			id = russia_events.9
            days = 5
		}
    }

    on_fail = { 
        trigger_event = {
             id = russia_events.3
            days = 5
        }
        random_civil_war = {
			add_civil_war_progress = -100
		}
    }

    invalid = {
        AND = {
          has_variable = entente_participant
          OR = {
           c:GER = {
             has_variable = central_powers_participant
		     has_modifier = country_humiliated
            } 
           c:AUS = {
             has_variable = central_powers_participant
		     has_modifier = country_humiliated
            } 
           }
        }
    }

    on_invalid = {
        random_civil_war = {
			add_civil_war_progress = -100
		}
        trigger_event = {
			id = russia_events.9
            days = 5
		}
	}

    weight = 1000
    should_be_pinned_by_default = yes
}

je_war_fatigue_russia = {
    icon = "gfx/interface/icons/event_icons/event_protest.dds"
    group = je_group_internal_affairs

    is_shown_when_inactive = {
        exists = c:RUS
        THIS = c:RUS
        has_variable = entente_participant
    }
    
    possible = {
        has_variable = entente_participant
        is_at_war = yes
        game_date > 1916.4.1
        root.average_sol < 10
    }

    immediate = {
       c:RUS = {
            trigger_event = {
                id = russia_events.2
                days = 5
            }
        }
    }

    modifiers_while_active = { war_protests_1_modifier }

    complete = { 
        OR = {
          has_law = law_type:law_council_republic
          root.average_sol >= 10 
          is_at_war = no
        }
    }

    invalid = {
        AND = {
          has_variable = entente_participant
           OR = {
             c:GER = {
                 has_variable = central_powers_participant
		         has_modifier = country_humiliated
                } 
             c:AUS = {
                 has_variable = central_powers_participant
		         has_modifier = country_humiliated
                } 
            }
        }
    }

    weight = 1000
    should_be_pinned_by_default = yes
}

join_entente_progress_bar = {
    name = "join_entente_progress_bar"
    desc = "join_entente_progress_bar_desc"

    default = yes 

    monthly_progress = {

        # Diplomatic Pacts
        owner = {
            every_country = {
                limit = {
                    has_variable = entente_participant

                add = { desc = "entente_progress_alliance_tt" value = 10 }
            }
            every_country = {
                limit = {
                    has_variable = entente_participant

                add = { desc = "entente_progress_defensive_pact_tt" value = 5 }
            }
            every_country = {
                limit = {
                    has_variable = entente_participant
                    has_diplomatic_pact = { who = root type = foreign_investment_agreement }
                }
                add = { desc = "entente_progress_investment_agreement_tt" value = 2 }
            }
            every_country = {
                limit = {
                    has_variable = entente_participant
                    has_diplomatic_pact = { who = root type = foreign_investment_rights }
                }
                add = { desc = "entente_progress_investment_rights_tt" value = 1 }
            }
            every_country = {
                limit = {
                    has_variable = entente_participant
                    has_diplomatic_pact = { who = root type = bankroll }
                }
                add = { desc = "entente_progress_bankroll_tt" value = 5 }
            }
            every_country = {
                limit = {
                    has_variable = entente_participant
                    has_diplomatic_pact = { who = root type = fund_lobbies }
                }
                add = { desc = "entente_progress_fund_lobbies_tt" value = 1 }
            }
            every_country = {
                limit = {
                    has_variable = central_powers_participant
                    has_diplomatic_pact = { who = root type = rivalry }
                }
                add = { desc = "entente_progress_from_rivalry_tt" value = 5 }
            }
        }

        # Lobbies
        owner = {
            every_political_lobby = {
                limit = {
                    is_political_lobby_type = lobby_pro_country
                    target = { has_variable = entente_participant }
                }
                add = { desc = "entente_progress_pro_lobby_tt" value = 3 }
            }
            every_political_lobby = {
                limit = {
                    is_political_lobby_type = lobby_anti_country
                    target = { has_variable = central_powers_participant }
                }
                add = { desc = "entente_progress_anti_lobby_tt" value = 3 }
            }
        }
    }

    start_value = 0
    min_value = 0
    max_value = 200
}

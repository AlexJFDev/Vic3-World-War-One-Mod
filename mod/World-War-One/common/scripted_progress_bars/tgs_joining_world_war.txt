join_entente_progress_bar = {
    name = "join_entente_progress_bar"
    desc = "join_entente_progress_bar_desc"

    default = yes 

    monthly_progress = { 
        #From trade
        if = {
		   limit = {
              any_country = {
                  has_variable = entente_participant
                  country_rank >= rank_value:major_power
                    has_diplomatic_pact = {
				      who = root
				      type = trade_agreement
			        }	
                }
            }
            add = {
				desc = "entente_progress_from_trade"
				value = 2
			}
        }
        #From diplomacy
        if = {
		   limit = {
              any_country = {
                  has_variable = entente_participant
                  country_rank >= rank_value:major_power
                    has_diplomatic_pact = {
				      who = root
				      type = foreign_investment_rights
			        }	
                }
            }
            add = {
				desc = "entente_progress_from_diplomacy"
				value = 1
			}
        }
        if = {
		   limit = {
              any_country = {
                  has_variable = entente_participant
                  country_rank >= rank_value:major_power
                    has_diplomatic_pact = {
				      who = root
				      type = foreign_investment_agreement
			        }	
                }
            }
            add = {
				desc = "entente_progress_from_diplomacy"
				value = 2
			}
        }
        if = {
		   limit = {
              any_country = {
                  has_variable = entente_participant
                  country_rank >= rank_value:major_power
                    has_diplomatic_pact = {
				      who = root
				      type = defensive_pact
			        }	
                }
            }
            add = {
				desc = "entente_progress_from_diplomacy"
				value = 10
			}
        }
        if = {
		   limit = {
              any_country = {
                  has_variable = entente_participant
                  country_rank >= rank_value:major_power
                    has_diplomatic_pact = {
				      who = root
				      type = alliance
			        }	
                }
            }
            add = {
				desc = "entente_progress_from_diplomacy"
				value = 20
			}
        }
        if = {
		   limit = {
              any_country = {
                  has_variable = central_powers_participant
                  country_rank >= rank_value:major_power
                    has_diplomatic_pact = {
				      who = root
				      type = rivalry
			        }	
                }
            }
            add = {
				desc = "entente_progress_from_diplomacy"
				value = 5
			}
        }
        if = {
		   limit = {
              any_country = {
                  has_variable = entente_participant
                  country_rank >= rank_value:major_power
                    has_diplomatic_pact = {
				      who = root
				      type = bankroll
			        }	
                }
            }
            add = {
				desc = "entente_progress_from_diplomacy"
				value = 8
			}
        }
        #From Lobbies
        if = {
		   limit = {
                any_political_lobby = {
			        is_political_lobby_type = lobby_pro_country
			        target = {
                        has_variable = entente_participant
                    }
	            }
            }
            add = {
				desc = "entente_progress_from_lobbies"
				value = 3
			}
        }
        if = {
		   limit = {
                any_political_lobby = {
			        is_political_lobby_type = lobby_anti_country
			        target = {
                        has_variable = entente_participant
                    }
	            }
            }
            add = {
				desc = "entente_progress_from_lobbies_2"
				value = -3
			}
        }
        #From decisions
        if = {
		   limit = {
                has_variable = supporting_central_powers_var
            }
            add = {
				desc = "entente_progress_from_decision"
				value = -5
			}
        }
        if = {
		   limit = {
                has_variable = supporting_entente_var
            }
            add = {
				desc = "entente_progress_from_decision"
				value = 10
			}
        }
        if = {
		   limit = {
                has_variable = maintain_neutrality_var
            }
            add = {
				desc = "entente_progress_from_decision"
				value = -10
			}
        }
    }

    start_value = 0
    min_value = 0
    max_value = 200
}
